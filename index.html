function actualizarPreview() { const d = getDatos(); document.getElementById('preview-pre').textContent = buildTexto(d); } function setStatus(msg, tipo='') { const el = document.getElementById('status-line'); el.innerHTML = (tipo==='ok' ? '' : tipo==='err'? '' : tipo==='busy'?'' : '') + msg; } async function accion(tipo) { const d = getDatos(); if (!validar(d)) return; const btnS = document.getElementById('btn-sheet'); const btnB = document.getElementById('btn-both'); btnS.disabled = btnB.disabled = true; if (tipo === 'sheet') { btnS.textContent = '⏳ Guardando...'; setStatus('Guardando en Sheet...','busy'); } if (tipo === 'ambos') { btnB.textContent = '⏳ Procesando...'; setStatus('Procesando...','busy'); } try { if (tipo === 'pdf') { await generarPDF(d); setStatus('PDF generado correctamente: ' + d.folio,'ok'); toast('PDF generado: ' + d.folio,'ok'); } else if (tipo === 'sheet') { await enviarSheet(d); setStatus('Guardado en Sheet: ' + d.folio,'ok'); toast('Registro guardado en Sheet.','ok'); } else if (tipo === 'ambos') { await generarPDF(d); await enviarSheet(d); setStatus('PDF + Sheet completado: ' + d.folio,'ok'); toast('PDF + Sheet completado correctamente.','ok'); } } catch(err) { setStatus('Error: ' + err.message,'err'); toast('Error: ' + err.message,'err'); } finally { btnS.disabled = btnB.disabled = false; btnS.textContent = '💾 Guardar en Sheet'; btnB.textContent = '⚡ PDF + Sheet'; } } function limpiar() { ['responsable','asunto','descripcion','resolucion'].forEach(id => document.getElementById(id).value = ''); document.getElementById('monto').value = ''; document.getElementById('categoria').value = ''; document.getElementById('emisor').value = 'Leonardo Castro'; document.querySelectorAll('#pg-estado .radio-opt').forEach(p => p.classList.remove('selected')); document.querySelector('#pg-estado .radio-opt').classList.add('selected'); document.getElementById('estado').value='Abierto'; document.querySelectorAll('#pg-prioridad .radio-opt').forEach(p => p.classList.remove('selected')); document.querySelectorAll('#pg-prioridad .radio-opt')[1].classList.add('selected'); document.getElementById('prioridad').value='Media'; const p = nowP(); document.getElementById('fecha').value = `${p.Y}-${p.M}-${p.D}`; document.getElementById('hora').value = `${p.h}:${p.m}:${p.s}`; document.getElementById('folio-display').textContent = buildFolio(p); setStatus('Listo',''); actualizarPreview(); } function toast(msg, tipo='ok') { const t = document.getElementById('toast'); const tit = document.getElementById('toast-title'); const bod = document.getElementById('toast-body'); tit.textContent = tipo==='err' ? 'Error del sistema' : 'AROMAKER – Sistema de Incidencias'; bod.textContent = msg; t.className = `toast95 ${tipo} show`; setTimeout(() => t.classList.remove('show'), 4000); }
